function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(data.sheet) || ss.insertSheet(data.sheet); if (data.action === 'append') { sheet.appendRow(data.row); return out({ok: true}); } if (data.action === 'read') { var rows = sheet.getLastRow() > 1 ? sheet.getRange(2, 1, sheet.getLastRow()-1, 9).getValues() : []; return out({rows: rows}); } if (data.action === 'delete') { var all = sheet.getDataRange().getValues(); for (var i = all.length; i >= 2; i--) { if (all[i-1][0] === data.id) sheet.deleteRow(i); } return out({ok: true}); } if (data.action === 'readAll') { var sheets = ['탄력시간발생','탄력시간사용']; var result = {}; sheets.forEach(function(name) { var s = ss.getSheetByName(name); result[name] = (s && s.getLastRow() > 1) ? s.getRange(2,1,s.getLastRow()-1,9).getValues() : []; }); return out(result); } } catch(err) { return out({error: err.message}); } } function out(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
